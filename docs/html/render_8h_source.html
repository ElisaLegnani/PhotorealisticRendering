109 struct PathTracer : public Renderer {
110 
111  PCG pcg;
112  int num_of_rays;
113  int max_depth;
114  int russian_roulette_limit;
115 
116  PathTracer(World w, Color bc = BLACK, PCG _pcg = PCG(), int nrays = 10, int maxd = 2, int rrlim = 3)
117  : Renderer(w, bc), pcg{_pcg}, num_of_rays{nrays}, max_depth{maxd}, russian_roulette_limit{rrlim} {}
118 
119  Color operator()(Ray ray) {
120 
121  if (ray.depth > max_depth)
122  return BLACK;
123 
124  HitRecord hit = world.ray_intersection(ray);
125 
126  if (!hit.init)
127  return background_color;
128 
129  Material hit_material = hit.material;
130  Color hit_color = hit_material.brdf->pigment->get_color(hit.surface_point);
131  Color emitted_radiance = hit_material.emitted_radiance->get_color(hit.surface_point);
132 
133  float hit_color_lum = fmax(fmax(hit_color.r, hit_color.g), hit_color.b);
134 
135  // Russian roulette
136  if (ray.depth >= russian_roulette_limit){
137  float q = fmax(0.05, 1. - hit_color_lum);
138  if (pcg.random_float() > q)
139  hit_color = hit_color * ( 1.0 / (1.0 - q));
140  // Keep the recursion going, but compensate for other potentially discarded rays
141  else
142  return emitted_radiance;
143  // Terminate prematurely
144  }
145 
146  Color cum_radiance(BLACK);
147  if (hit_color_lum > 0.0){ // Only do costly recursions if it's worth it
148  for (int i{}; i < num_of_rays; ++i){
149  Ray new_ray = hit_material.brdf->scatter_ray(pcg, hit.ray.dir, hit.world_point, hit.normal, (ray.depth + 1));
150  // Recursive call
151  Color new_radiance = (*this)(new_ray);
152  cum_radiance = cum_radiance + hit_color * new_radiance;
153  }
154  }
155 
156  return emitted_radiance + cum_radiance * (1.0 / num_of_rays);
157  }
158 };
